# Allow read only from these directories (local or NAS mount)

# ===== Gateway Contracts (suggested) =====

## 1) Image Generate

{
  "provider": "openai|stability|replicate|custom",

  "negativePrompt": "...",
  "size": "1024x1024",
  "n": 1,
  "seed": 123,
  "outputFormat": "png|jpg|webp",
  "returnMode": "file|base64"
}

{
  "images": [{"b64":"...","mime":"image/png"}],
  "meta": {...}
}

Response (option B - file path):
{
  "filePath": "/abs/path/or/relative/path.png",
  "meta": {...}
}

{
  "url": "https://.../image.png",
  "meta": {...}
}

POST /v1/list

{
  "provider":"gdrive|onedrive|nas",

  "maxItems":200
}

{ "items":[ { "name":"...", "idOrPath":"...", "mime":"...", "size":123, "modifiedAt":"..." } ] }
